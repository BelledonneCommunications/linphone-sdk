A most simple exmaple to test out whether CUnit is properly working or not would be to just write some skeleton functions and checkout. It could be something as follows,

#include <stdio.h>
#include <stdlib.h>
#include <CUnit/Console.h>

int init_group1(void)
{
   return 0;
}

int clean_group1(void)
{
   return 0;
}

void testSuccess(void)
{
   Assert(1);
}

void testFailure(void)
{
   Assert(0);
}

int main(void)
{
   PTestGroup pGroup = NULL;
   PTestCase pTest = NULL;
   if (initialize_registry())
      return 1;
   pGroup = add_test_group("Group 1",init_group1, clean_group1);
   if (!pGroup) {
      cleanup_registry();
      return 2;
   }
   pTest = add_test_case(pGroup, "testSuccess",testSuccess);
   if (!pTest) {
      cleanup_registry();
      return 3;
   }
   pTest = add_test_case(pGroup, "testFailure",testFailure);
   if (!pTest) {
      cleanup_registry();
      return 3;
   }
   /* call console_run_tests() for Console based interface
    and curses_run_tests() for Curses based interface */
   console_run_tests();
   cleanup_registry();
   return 0;
}
